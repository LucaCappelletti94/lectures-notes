\section{Probabilità di rientro per stati transienti}
\begin{theorem}[Probabilità di rientro per stati transienti]
  Per ogni \(i,j \in S\), se \(j\) è transiente allora:
  \[
    P^{(n)}_{ij} = \O{\epsilon^n}
  \]
  Per qualche \(0<\epsilon<1\).
\end{theorem}
