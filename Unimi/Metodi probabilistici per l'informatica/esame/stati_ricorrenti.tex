\section{Stati ricorrenti}
\begin{theorem}[Stati ricorrenti]
  Per ogni stato \(i \in S\) le seguenti proprietà sono equivalenti:
  \begin{enumerate}
    \item \(i\) è ricorrente
    \item \(f_{ii} = 1\)
    \item \(\sum_{n \geq 0}p^{(n)}_{ii} = + \infty \)
  \end{enumerate}
\end{theorem}
