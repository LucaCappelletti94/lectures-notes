\section{Il duale del duale è il primale}

\begin{theorem}
  Il duale del problema duale è il problema primale.
\end{theorem}

\begin{proof}
  Considero un problema $P$ in forma canonica e procedo con le seguenti operazioni:

  \begin{enumerate}
    \item Trasformo $P$ nel duale $D$.
    \item Riscrivo $D$ in modo da ``assomigliare'' a $P$.
    \item Trasformo il $D$ riscritto in $P'$.
    \item Riscrivo $P'$ ed esso coincide con $P$.
  \end{enumerate}

  \[
    \begin{cases}
      \min \bmct \bmx \\
      A\bmx \geq \bmb \\
      \bmx \geq \bm{0}
    \end{cases}
    \Rightarrow
    \begin{cases}
      \max \bmut\bmb     \\
      \bmct \geq \bmut A \\
      \bmu \geq \bm{0}
    \end{cases}
    \equiv
    \begin{cases}
      - \min (-\bmbt)\bmu   \\
      (-A^T)\bmu \geq -\bmc \\
      \bmu \geq \bm{0}
    \end{cases}
    \Rightarrow
    \begin{cases}
      - \max \bmy^T(-\bmc)     \\
      -\bmbt \geq \bmy^T(-A^T) \\
      \bmy \geq \bm{0}
    \end{cases}
    \equiv
    \begin{cases}
      \min \bmct \bmy \\
      A\bmy\geq \bmb  \\
      \bmy \geq \bm{0}
    \end{cases}
  \]

\end{proof}
