\chapter{Introduzione}
\begin{definition}[Path (Cammino)]
  Una sequenza di \textbf{vertici} \(C = \crl{x_0, \ldots, x_n}\) con \(x_i \in V\), tale che:
  \[
    \forall x_i, x_{i+1} \in C \quad x_i \rightarrow x_{i+1}
  \]
\end{definition}
